function computeSteps(a) { const s = []; const data = [...a]; const n = data.length; for (let i = 0; i < n - 1; i++) { for (let j = 0; j < n - i - 1; j++) { s.push({ arr: [...data], i, j, msg: `Comparing ${data[j]} and ${data[j+1]}`, type: 'comp' }); if (data[j] > data[j + 1]) { [data[j], data[j + 1]] = [data[j + 1], data[j]]; s.push({ arr: [...data], i, j, msg: `Swapping ${data[j+1]} > ${data[j]}`, type: 'swap' }); } } } s.push({ arr: [...data], msg: "Sorted Successfully!", type: 'done' }); return s; }
